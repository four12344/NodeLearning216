	npm
		查看版本
			npm -v
		安装模块
			npm install <Module Name> -g
				加-g表全局安装 不加则表示本地安装
		查看全局安装的模块
			npm list -g
		查看某个模块的版本号
			npm list express
		卸载模块
			npm uninstall express
		更新模块
			npm update express
		搜索模块
			npm search express
	node的终端
		学名：REPL交互式解释器
		使用方法
			打开
				终端输入：node
			关闭
				当前
					^ c
				整个终端
					^c ^c
